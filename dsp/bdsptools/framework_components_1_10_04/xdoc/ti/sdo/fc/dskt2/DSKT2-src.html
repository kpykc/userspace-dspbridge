    37        /*!
    38         *  ======== DARAM2 ========

    40         */
    41        config string DARAM2     = DEFAULTSEG;
    42    
    43        /*!
    44         *  ======== SARAM0 ========
    45         *  IALG memory space mapping of IALG_SARAM0.  This must match a BIOS heap.
    46         */
    47        config string SARAM0     = DEFAULTSEG;
    48    
    49        /*!
    50         *  ======== SARAM1 ========
    51         *  IALG memory space mapping of IALG_SARAM1.  This must match a BIOS heap.

    55        /*!
    56         *  ======== SARAM2 ========
    57         *  IALG memory space mapping of IALG_SARAM2.  This must match a BIOS heap.

    59        config string SARAM2     = DEFAULTSEG;
    60    
    61        /*!
    62         *  ======== ESDATA ========
    63         *  IALG memory space mapping of IALG_ESDATA.  This must match a BIOS heap.

    65        config string ESDATA     = DEFAULTSEG;
    66    
    67        /*!
    68         *  ======== IPROG ========
    69         *  IALG memory space mapping of IALG_IPROG.  This must match a BIOS heap.
    70         */
    71        config string IPROG      = DEFAULTSEG;
    72    
    73        /*!
    74         *  ======== EPROG ========
    75         *  IALG memory space mapping of IALG_EPROG.  This must match a BIOS heap.
    76         */
    77        config string EPROG      = DEFAULTSEG;
    78    
    79    
    80        /*!
    81         *  ======== DSKT2_HEAP ========
    82         *  Memory segment for internal DSKT2 data structures.

    84         *  Objects internal to the DSKT2 module will be allocated from this heap.
    85         */
    86        config string DSKT2_HEAP = DEFAULTSEG;
    87    
    88        /*!
    89         *  ======== ALLOW_EXTERNAL_SCRATCH ========
    90         *  Allow external memory to be used for algorithm scratch memory requests.
    91         *
    92         *  If an algorithm requests scratch memory, and insufficient scratch
    93         *  memory is available, scratch buffers will be allocated in external

    95         */

    99         *  ======== DARAM_SCRATCH_SIZES ========

   111    
   112        /*!

   131         *  Rules and Guidelines.
   132         */

   134    
   135    }
